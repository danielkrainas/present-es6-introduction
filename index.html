Diving into the Shallows

Let there be const!

Keywords for defining block-scoped variables


const foo = 1;
foo = 2; //error: cannot change the value of a constant

{
	console.log(hi); //error, the variable is not defined
	//this is only available within this block
	let hi = 1;
}

console.log(hi); //error, hi is not defined
for(let i = 0; i < 10; i++) {
	//the variable `i` is only available within the for block
}

console.log(i); //error, i is not defined

Computed Properties

A cleaner syntax for defining dynamic properties


// old and busted
var obj = { };
obj[prop] = 'some value';

// new hotness
var obj = { [prop]: 'some value' };

Default Parameters

A native and cleaner method of defaulting function parameters


// old and busted
function helloOld(name) {
	name = name || 'Anonymous';
  	console.log('Hi ' + name);
}

// new hotness
function hello(name = 'Anonymous') {
  	console.log('Hi ' + name);
}

Destructuring Assignments

Short hand sugar for accessing object properties


// old and busted
var name = user.name;
var email = user.email;

// new hotness
var { name, email } = user;

// even hotter
function complex({ name, email, phone = null }) {
  //variables `name`, `email`, and `phone` are available here
}
 
var user = {
  name: 'foo',
  email: 'bar',
  phone: '123456'
};
complex(user);

Object Initializer Shorthand


// old and busted
module.exports = {
  someFunction: someFunction,
  otherFunction: otherFunction
};

// new hotness
module.exports = {
  someFunction,
  otherFunction
};

Rest and Spread Parameters


// old and busted
function push() {
  var args = Array.prototype.slice.call(arguments);
  var list = args.shift();
 
  args.forEach(function(a) {
    list.push(a);
  });
}

// new hotness
function push(list, ...values) {
  values.forEach(function(v) {
    list.push(v);
  });
}

// variable spreading
values = [1,2,3];
var target = [];
target.push(...values);

Templates


// old and busted
var name = "Björn";
var text = "Hello " + name;

// new hotness
var name = "Björn";
var text = `Hello ${name}`;

I Promise, No Pyramids


// old and busted
function asyncA(callback) { setTimeout(callback, 100); }
function asyncB(callback) { setTimeout(callback, 200); }
function asyncC(callback) { setTimeout(callback, 300); }
 
asyncA(function(resultA) {
  asyncB(function(resultB) {
    asyncC(function(resultC) {
      someFunction(resultA, resultB, resultC);
    });
  });
});

// new hotness
function promiseA() { return new Promise((resolve, reject) => setTimeout(resolve, 100)); }
function promiseB() { return new Promise((resolve, reject) => setTimeout(resolve, 200)); }
function promiseC() { return new Promise((resolve, reject) => setTimeout(resolve, 300)); }
 
Promise.all([promiseA(), promiseB(), promiseC()]).then(([a, b, c]) => {
  someFunction(a, b, c);
});

Arrows Arrows Everywhere

Implicit binding to outside scope without bind


//arrow function with no parameters
var a1 = () => 1;
 
//arrow with one parameter can be defined without parentheses
var a2 = x => 1;
var a3 = (x) => 1;
 
//arrow with multiple params requires parentheses
var a4 = (x, y) => 1;
 
//arrow with body has no implicit return
var a5 = x => { return 1; };

Harmonious Modules


// lib.js
export const sqrt = Math.sqrt;
export function square(x) {
    return x * x;
}

export function diag(x, y) {
    return sqrt(square(x) + square(y));
}

//------ main.js ------
import { square, diag } from 'lib';
console.log(square(11)); // 121
console.log(diag(4, 3)); // 5
